"""
Staged Persona Service - Two-stage LLM calls to prevent information dumping

Stage 1: Determine sharing boundaries (what character willing to reveal)
Stage 2: Generate actual response within those boundaries
"""

import json
import logging
from typing import Dict, List, Optional, Any
from datetime import datetime

from src.services.llm_service import LLMService
from src.models.conversation_state import ConversationStateManager
from src.config.settings import get_settings
from src.dependencies import get_supabase_client

logger = logging.getLogger(__name__)

class StagedPersonaService:
    """
    Multi-stage persona response generation
    Prevents personas from volunteering all backstory at once
    """
    
    def __init__(self):
        self.llm_service = LLMService()
        self.settings = get_settings()
        self.supabase = get_supabase_client()
        self.state_manager = ConversationStateManager(self.supabase)
    
    async def process_conversation(
        self,
        user_message: str,
        persona_id: str,
        session_id: str,
        conversation_history: Optional[List[Dict]] = None,
        persona_llm: str = "gpt-4o-mini"
    ) -> Dict[str, Any]:
        """
        Process conversation using two-stage approach
        
        Returns dict with keys: response, trust_level, empathy_score, stage, is_safe
        """
        try:
            # Get persona base data from personas table
            persona_data = await self._get_persona(persona_id)
            persona_name = persona_data.get('name', 'Unknown')
            
            # Get current conversation state (includes trust level)
            conversation_state = await self.state_manager.get_conversation_state(session_id)
            trust_level = getattr(conversation_state, 'trust_level', 3.0)  # Default to defensive
            turn_count = getattr(conversation_state, 'turn_count', 0)
            
            logger.info(f"Processing turn {turn_count + 1} for {persona_name}, trust={trust_level:.2f}")
            
            # Analyze user message empathy using gpt-4o-1-nano
            mi_analysis = await self._analyze_empathy(user_message, conversation_history or [])
            empathy_score = mi_analysis.get('empathy_score', 5)
            
            # Get available knowledge for current trust level
            available_knowledge = await self._get_available_knowledge(persona_id, trust_level)
            
            # STAGE 1: Determine sharing boundaries
            logger.info(f"Stage 1: Determining sharing boundaries (trust={trust_level:.2f})")
            sharing_boundaries = await self._determine_sharing_boundaries(
                persona_data=persona_data,
                available_knowledge=available_knowledge,
                user_message=user_message,
                trust_level=trust_level,
                empathy_score=empathy_score,
                session_id=session_id,
                turn_number=turn_count + 1
            )
            
            # STAGE 2: Generate actual response using boundaries
            logger.info(f"Stage 2: Generating response within boundaries")
            response = await self._generate_bounded_response(
                persona_data=persona_data,
                sharing_boundaries=sharing_boundaries,
                user_message=user_message,
                conversation_history=conversation_history or [],
                persona_llm=persona_llm,
                session_id=session_id,
                turn_number=turn_count + 1,
                trust_level=trust_level
            )
            
            # Update conversation state (trust level changes based on empathy)
            updated_state = await self.state_manager.update_conversation_state(
                session_id, mi_analysis, user_message
            )
            
            # Log to conversation_transcripts (existing table)
            await self._log_to_transcript(
                session_id=session_id,
                user_message=user_message,
                persona_response=response.strip(),
                empathy_score=empathy_score
            )
            
            logger.info(f"Turn complete. New trust: {updated_state.trust_level:.2f}")
            
            return {
                'response': response.strip(),
                'trust_level': updated_state.trust_level,
                'empathy_score': empathy_score,
                'stage': updated_state.stage,
                'is_safe': mi_analysis.get('is_safe', True)
            }
            
        except Exception as e:
            logger.error(f"Staged persona processing failed: {e}", exc_info=True)
            raise
    
    async def _get_persona(self, persona_id: str) -> Dict:
        """Get persona base data from personas table"""
        result = self.supabase.table('personas').select('*').eq(
            'persona_id', persona_id
        ).maybe_single().execute()
        
        if not result.data:
            raise RuntimeError(f"Persona '{persona_id}' not found in database")
        
        return result.data
    
    async def _get_available_knowledge(self, persona_id: str, trust_level: float) -> Dict:
        """
        Get knowledge available at current trust level
        Returns highest tier where trust_threshold <= current trust
        """
        result = self.supabase.table('character_knowledge_tiers').select('*').eq(
            'persona_id', persona_id
        ).lte('trust_threshold', trust_level).order(
            'trust_threshold', desc=True
        ).limit(1).execute()
        
        if not result.data:
            logger.warning(f"No knowledge tier found for {persona_id} at trust {trust_level:.2f}, using defensive fallback")
            return {
                "job_title": "employee",
                "context": "performance review",
                "stance": "guarded"
            }
        
        tier = result.data[0]
        tier_name = tier.get('tier_name', 'unknown')
        knowledge = tier.get('knowledge_json', {})
        
        logger.info(f"Using tier '{tier_name}' (threshold {tier['trust_threshold']}) with {len(knowledge)} knowledge items")
        
        return knowledge
    
    async def _determine_sharing_boundaries(
        self,
        persona_data: Dict,
        available_knowledge: Dict,
        user_message: str,
        trust_level: float,
        empathy_score: int,
        session_id: str,
        turn_number: int
    ) -> str:
        """
        STAGE 1: Use fast LLM to determine what character would share
        
        This is a META analysis - determines boundaries without generating response
        """
        persona_name = persona_data.get('name', 'Person')
        
        # Build Stage 1 prompt
        stage1_prompt = f"""You are analyzing what a character in a performance review would be willing to share.

CHARACTER: {persona_name}
TRUST LEVEL: {trust_level:.1f}/10.0 (scale: 3.0=defensive, 5.5=cautious, 7.5=opening, 9.0=trusting)
USER'S EMPATHY THIS TURN: {empathy_score}/10

FACTS CHARACTER KNOWS AT THIS TRUST LEVEL:
{json.dumps(available_knowledge, indent=2)}

USER JUST SAID: "{user_message}"

TASK: Determine what {persona_name} would be WILLING to share in response.

GUIDELINES:
- Trust 3.0-5.4 (defensive): Share only surface facts, deflect personal questions politely
- Trust 5.5-7.4 (cautious): Hint at issues without specifics (e.g., "family stuff", "personal matters")
- Trust 7.5-8.9 (opening): Share specific details if asked with empathy
- Trust 9.0+ (trusting): Share vulnerabilities openly

Respond with valid JSON only:
{{
  "willing_to_mention": ["topic1", "topic2"],
  "deflection_needed": true/false,
  "specificity_level": "vague"/"hints"/"detailed",
  "emotional_stance": "guarded"/"cautious"/"open"/"vulnerable",
  "reasoning": "brief explanation of why"
}}"""
        
        # Call fast model for Stage 1
        stage1_response = await self.llm_service.generate_response(
            prompt=stage1_prompt,
            model="gpt-4o-mini",
            temperature=0.3,
            max_tokens=300
        )
        
        # Log Stage 1
        await self._log_stage(
            session_id=session_id,
            turn_number=turn_number,
            stage_number=1,
            stage_name='boundary_determination',
            llm_model='gpt-4o-mini',
            input_prompt=stage1_prompt,
            output_text=stage1_response,
            trust_level=trust_level
        )
        
        logger.debug(f"Stage 1 output: {stage1_response[:200]}...")
        
        return stage1_response
    
    async def _generate_bounded_response(
        self,
        persona_data: Dict,
        sharing_boundaries: str,
        user_message: str,
        conversation_history: List[Dict],
        persona_llm: str,
        session_id: str,
        turn_number: int,
        trust_level: float
    ) -> str:
        """
        STAGE 2: Generate actual response using determined boundaries
        
        Persona LLM sees boundaries but NOT raw knowledge facts
        """
        persona_name = persona_data.get('name', 'Person')
        system_context = persona_data.get('system_context', '')
        
        # Build recent conversation context
        recent_context = ""
        if conversation_history:
            last_n = 32 if len(conversation_history) > 50 else 16
            last_exchanges = conversation_history[-last_n:]
            recent_context = "\n".join([
                f"{'User' if msg.get('role') == 'user' else 'You'}: {msg.get('content', '')}"
                for msg in last_exchanges
            ])
        
        # Parse Stage 1 boundaries
        try:
            boundaries = json.loads(sharing_boundaries.strip().replace('```json', '').replace('```', ''))
        except:
            logger.warning("Failed to parse Stage 1 JSON, using fallback boundaries")
            boundaries = {
                "willing_to_mention": [],
                "deflection_needed": False,
                "specificity_level": "vague",
                "emotional_stance": "cautious"
            }
        
        # Build Stage 2 prompt
        stage2_prompt = f"""{system_context}

RECENT CONVERSATION:
{recent_context}

[Internal guidance - do not reference explicitly]:
- Willing to mention: {', '.join(boundaries.get('willing_to_mention', ['nothing specific']))}
- Specificity: {boundaries.get('specificity_level', 'vague')}
- Emotional stance: {boundaries.get('emotional_stance', 'cautious')}
- Should deflect: {boundaries.get('deflection_needed', False)}

User just said: "{user_message}"

Respond naturally as {persona_name}. Stay true to your emotional stance. Do not volunteer information you're not ready to share."""
        
        # Call persona LLM for Stage 2
        persona_model = self._normalize_model_name(persona_llm)
        stage2_response = await self.llm_service.generate_response(
            prompt=stage2_prompt,
            model=persona_model,
            temperature=0.8,
            max_tokens=500,
            conversation_history=conversation_history
        )
        
        # Log Stage 2
        await self._log_stage(
            session_id=session_id,
            turn_number=turn_number,
            stage_number=2,
            stage_name='response_generation',
            llm_model=persona_model,
            input_prompt=stage2_prompt,
            output_text=stage2_response,
            trust_level=trust_level
        )
        
        return stage2_response
    
    async def _analyze_empathy(
        self,
        user_message: str,
        conversation_history: List[Dict]
    ) -> Dict[str, Any]:
        """
        Analyze MI technique using gpt-4o-1-nano
        
        Returns dict with empathy_score (1-10), techniques_used, etc.
        """
        analysis_prompt = f"""Analyze this MI interaction and respond with JSON:

{{
  "empathy_score": 1-10,
  "techniques_used": ["reflection", "open_question", "affirmation"],
  "openness_change": "increase"/"same"/"decrease",
  "is_safe": true/false
}}

MI SCORING (BE STRICT):
- 1-3: Hostile, judgmental, advice-giving
- 4-5: Basic questions, no empathy
- 6-7: Open questions + reflection
- 8-9: Deep reflection + affirmation
- 10: Perfect MI technique

User message: "{user_message}" """
        
        try:
            response = await self.llm_service.generate_response(
                prompt=analysis_prompt,
                model="gpt-4o-1-nano-2025-04-14",
                temperature=0.3,
                max_tokens=150
            )
            
            analysis = json.loads(response.strip().replace('```json', '').replace('```', ''))
            
            return {
                'empathy_score': analysis.get('empathy_score', 5),
                'techniques_used': analysis.get('techniques_used', []),
                'openness_change': analysis.get('openness_change', 'same'),
                'is_safe': analysis.get('is_safe', True)
            }
        except Exception as e:
            logger.warning(f"MI analysis failed: {e}")
            return {
                'empathy_score': 5,
                'techniques_used': ['unknown'],
                'openness_change': 'same',
                'is_safe': True
            }
    
    async def _log_stage(
        self,
        session_id: str,
        turn_number: int,
        stage_number: int,
        stage_name: str,
        llm_model: str,
        input_prompt: str,
        output_text: str,
        trust_level: float
    ):
        """Log each stage to staged_generation_log table"""
        try:
            log_data = {
                'session_id': session_id,
                'turn_number': turn_number,
                'stage_number': stage_number,
                'stage_name': stage_name,
                'llm_model': llm_model,
                'input_prompt': input_prompt,
                'output_text': output_text,
                'trust_level': trust_level,
                'created_at': datetime.now().isoformat()
            }
            
            self.supabase.table('staged_generation_log').insert(log_data).execute()
        except Exception as e:
            logger.error(f"Failed to log stage: {e}")
    
    async def _log_to_transcript(
        self,
        session_id: str,
        user_message: str,
        persona_response: str,
        empathy_score: int
    ):
        """
        Log to existing conversation_transcripts table
        
        IMPORTANT: Check the actual schema of conversation_transcripts table.
        Adjust field names as needed based on your actual table structure.
        """
        try:
            # This is a guess at your schema - adjust as needed
            transcript_data = {
                'session_id': session_id,
                'user_message': user_message,

            "gpt": "gpt-5-chat-latest",
            "gemini": "gemini-2.5-flash",
            "google": "gemini-2.5-flash"
        }
        return model_mapping.get(persona_llm.lower(), "gpt-5-chat-latest")
